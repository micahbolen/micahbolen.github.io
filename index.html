# Write two Python functions to find the minimum number in a list.

# The first function should compare each number to every other number on the
# list. O(n2).
def compareNumbers(numberList):
    print('Comparing numbers...')
    print(numberList)

    i = 0
    j = 1

    minimumNumber = numberList[0]

    for number1 in numberList:
        if(j < len(numberList) - 1):
            number2 = numberList[j]

        if(i == 0):
            minimumNumber = number1

        if(j % 2 == 0):
            print('Comparing ' + str(number1) + ' and ' + str(number2) + '...')
            minimumInCurrentPair = findMinimum(number1, number2)
            print('The minimum number in the list currently is: ')
            print(minimumNumber)
            if(minimumInCurrentPair < minimumNumber):
                minimumNumber = minimumInCurrentPair

        j += 1
        i += 1

    print('The minimum number in the entire list is: ')
    return minimumNumber

# The second function should be linear O(n).
def findMinimum(firstNumber, secondNumber):
    if(firstNumber > secondNumber) :
        print('The first number is greater than the second number.')
        return secondNumber
    elif(firstNumber < secondNumber):
        print('The first number is less than the second number.')
        return firstNumber
    else :
        print('The first number is equal to the second number.')
        return firstNumber

The Infinite Monkey Theorem


from __future__ import division
from random import randrange

# The infinite monkey theorem states that a monkey hitting keys at random on a
# typewriter keyboard for an infinite amount of time will almost surely type a
# given text, such as the complete works of William Shakespear.  Well, suppose
# we replace a monkey with a Python function.  How long do you think it would
# take for a Python function to generate just one sentence of Shakespeare?  The
# sentence we'll shoot for is: "methinks it is like a weasel"

# Write a function that generates a string that is 28 characters long by
# choosing random letters from the 26 letters in the alphabet plus the space.
def generate():
  stringLength = 28
  stringList = list(range(0, stringLength))
  letters = 'abcdefghijklmnopqrstuvwxyz'
  space = ' '
  seedList = list(letters + space)

  position = 0

  while position < stringLength:
    stringList[position] = seedList[randrange(0, len(seedList))]
    position += 1

  string = ''.join(stringList)

  return string

# Write another function that will score each generated string by comparing the
# randomly generated string to the goal.
def score(submission):
  score = 0
  matchingCharacters = 0
  goal = 'methinks it is like a weasel'

  position = 0

  while position < len(goal):
    if submission[position] == goal[position]:
      matchingCharacters += 1

    position += 1

  # since we are using Python < 3, we need to import the division module
  # to make sure that integer division results in a floating point value
  score = (matchingCharacters / len(goal)) * 100

  return score


# Write a third function that will repeatedly call generate and score, then if
# 100% of the letters are correct we are done.  If the letters are not correct
# then we will generate a whole new string.  To make it easer to follow your
# program's progress this third function should print out the best string
# generated so far and its score every 1000 tries.
def test(maxTries = 20000):
  progress = 0
  bestScoreSoFar = 0

  while progress < maxTries:
    testString = generate()
    testScore = score(testString)
    progress += 1

    if testScore > bestScoreSoFar:
      bestScoreSoFar = testScore

      return

      print('Progress: %i strings tested so far' % progress)
      print('Best score so far: %f' % bestScoreSoFar)
